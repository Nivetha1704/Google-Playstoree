# Convert 'Size' to numeric (e.g. "19M" -> 19.0, "1.2k" -> 1.2)
def size_to_mb(size):
    if pd.isna(size):
        return None
    if "M" in size:
        return float(size.replace("M", ""))
    if "k" in size:
        return float(size.replace("k", "")) / 1024
    return None

df["Size"] = df["Size"].apply(size_to_mb)
